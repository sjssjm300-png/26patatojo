#include // 핀 설정 const int sensorPin = A0; // 이지메이커 가스 센서 const int servoPin = 9; // 서보모터 (출입구 제어) // 필터 및 데이터 관련 변수 float alpha = 0.1; // EMA 필터 계수 (낮을수록 부드러움) float filteredValue = 0; float Ro = 10.0; // 깨끗한 공기 기준 저항값 // 출입구 제어 변수 Servo myServo; bool isOpen = false; const int openAngle = 90; // 문 열림 각도 const int closeAngle = 0; // 문 닫힘 각도 void setup() { Serial.begin(9600); myServo.attach(servoPin); myServo.write(closeAngle); // 시작 시 문 닫음 filteredValue = analogRead(sensorPin); Serial.println("Time(s),Voltage,Rs_Ro_Ratio,Gate_Status"); } void loop() { // 1. 센서 값 읽기 및 EMA 필터 (방방 뜨는 값 보정) int rawRead = analogRead(sensorPin); filteredValue = (rawRead * alpha) + (filteredValue * (1.0 - alpha)); // 2. 물리적 수치 변환 float voltage = filteredValue * (5.0 / 1023.0); float Rs = ((5.0 * 10.0) / (voltage + 0.001)) - 10.0; // 0나누기 방지 float ratio = Rs / Ro; // 3. 출입구 제어 로직 (예시: 농도 전압이 3.0V를 넘으면 환기 위해 개방) // 실험 목적에 따라 "if (millis() > 60000)" 처럼 시간제로 바꿔도 됩니다. if (voltage > 3.0 && !isOpen) { myServo.write(openAngle); isOpen = true; } else if (voltage <= 1.5 && isOpen) { myServo.write(closeAngle); isOpen = false; } // 4. 시리얼 모니터 출력 (엑셀 데이터용) Serial.print(millis() / 1000); Serial.print(","); Serial.print(voltage); Serial.print(","); Serial.print(ratio); Serial.print(","); Serial.println(isOpen ? "OPEN" : "CLOSED"); delay(1000); // 1초 단위 로깅 }
